### This file will define the deck class and also run the 4 decks based on the different houses. 


# import python packages
import random 

# import from other py files
from card import Card


# class: deck 
class DeckClass():    

    def __init__(self, house_dict, house):
        '''There will be 4 decks/instances be based on the house,
        which in itself has different anecdote and value dictionary.'''
        self.house_dict = house_dict
        self.house = house
        self.deck_list = []
        self.build()
        
    # method: build 
    def build(self):
        '''Build the deck for one Hogwarts House.'''
        for key, value in self.house_dict.items():
            self.deck_list.append(Card(key, value, self.house))
        return (self.deck_list)

    
    # method: print_deck 
    def print_deck(self):
        '''Have a function to print the deck - more for development purposes.'''
        for cc in self.deck_list:
            Card.card_print(cc)
            
    # method: shuffle_deck 
    def shuffle_deck(self):
        '''Shuffle the deck in question.'''
        random.shuffle(self.deck_list)
    
    # method: draw_card
    def draw_card(self):
        '''Draw 1 card which will be used to help move the 
        player on the board'''
        card_drawn = self.deck_list.pop()
        return (card_drawn)
    
    # method: len_deck
    def len_deck(self):
        '''Get the number of cards remaining in the deck'''
        return len(self.deck_list)
    
    # method: deck_empty
    def deck_empty(self):
        '''Find out if the deck is empty. This will be used to 
        inform whether the player has lost the game.'''
        if len(self.deck_list) == 0:
            return True
        else:
            return False 

    def __repr__(self):
        return (self.deck_list)
